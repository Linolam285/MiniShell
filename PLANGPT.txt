│

│   ├─ main.cpp           # point d’entrée, boucle principale
│   ├─ Shell.cpp          # implémentation de la classe Shell
│   ├─ Command.cpp        # implémentation des commandes
│   ├─ Builtins.cpp       # implémentation des built-ins
│   └─ Utils.cpp          # implémentation des fonctions utilitaires
│
├─ tests/                 # tests unitaires ou scripts de test
│   └─ test_shell.cpp     
│
├─ CMakeLists.txt         # pour construire le projet avec CMake
├─ README.md              # description du projet, instructions
